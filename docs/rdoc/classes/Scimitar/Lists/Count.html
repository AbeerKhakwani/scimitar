def limit=(value)
  return if value.blank? # NOTE EARLY EXIT

  validate_numericality(value)
  input = value.to_s.to_i
  raise if input < 1
  @limit = input

def start_index=(value)
  return if value.blank? # NOTE EARLY EXIT

  validate_numericality(value)
  input = value.to_s.to_i
  input = 1 if input < 1
  @start_index = input
end
