add_custom_resource(resource) Link

Can be used to add a new resource type which is not provided by the gem. For example:

module Scim
  module Resources
    class ShinyResource < Scimitar::Resources::Base
      set_schema Scim::Schema::Shiny

      def self.endpoint
        "/Shinies"
      end
    end
  end
end
